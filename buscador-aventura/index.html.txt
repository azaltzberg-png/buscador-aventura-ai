<script>
    let map;
    let markers = [];
    const SERVER_TOKEN = '5e8c97e2f82bfeedf093b27b03c8acbd';

    function initMap() {
        map = new google.maps.Map(document.getElementById("map"), {
            center: { lat: 25.9565, lng: -80.1392 },
            zoom: 14,
            mapId: 'da9b6acc1483c1e535943ad2'
        });
    }

    async function fetchMLS() {
        const resultsDiv = document.getElementById('results');
        resultsDiv.innerHTML = "Cargando propiedades reales...";
        markers.forEach(m => m.setMap(null));

        try {
            // Usamos un proxy para que el MLS no bloquee la conexión
            const target = `https://api.bridgedataoutput.com/api/v2/OData/miami/Property?access_token=${SERVER_TOKEN}&$filter=City eq 'Aventura' and MlsStatus eq 'Active'&$top=15`;
            const response = await fetch("https://api.allorigins.win/get?url=" + encodeURIComponent(target));
            const json = await response.json();
            const data = JSON.parse(json.contents);

            resultsDiv.innerHTML = "";
            data.value.forEach(prop => {
                const pos = { lat: prop.Latitude, lng: prop.Longitude };
                const marker = new google.maps.Marker({ position: pos, map: map });
                markers.push(marker);

                const card = document.createElement('div');
                card.className = 'card';
                card.innerHTML = `
                    <img src="${prop.Media?.[0]?.MediaURL || 'https://via.placeholder.com/300x150'}">
                    <div style="font-weight:bold; color: #c5a059; font-size: 1.1rem; margin-top:5px;">$${prop.ListPrice.toLocaleString()}</div>
                    <div style="font-size:13px; font-weight:600;">${prop.UnparsedAddress}</div>
                `;
                card.onclick = () => { map.setZoom(17); map.panTo(pos); };
                resultsDiv.appendChild(card);
            });
        } catch (e) {
            resultsDiv.innerHTML = "Error al conectar con el MLS.";
        }
    }
</script>
